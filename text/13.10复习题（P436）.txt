1. 下面的程序有什么问题？
    int main(void)
    {
        int * fp;
        int k;

        fp = fopen("gelation");
        for (k = 0; k < 30; k++)
            fputs(fp, "Nanette eats gelation.");
        fclose("gelation");
        return 0;
    }

    问题：
    1) fp是一个整形指针，而不是一个文件指针。
    2) fopen()需要选择文件打开模式。
    3) fclose()函数需要一个文件指针，而不是一个字符串。

2. 下面的程序完成什么任务？（假设在命令行环境中运行）
    #include <stdio.h>
    #include <stdlib.h>
    #include <ctype.h>

    int main(int argc, char * argv[])
    {
        int ch;
        FILE * fp;
        if (argc < 2)
            exit(EXIT_FAILURE);
        if ((fp = fopen(argv[1], "r")) == NULL)
            exit(EXIT_FAILURE);
        while((ch = getc(fp)) != EOF)
            if (isdigit(ch))
                putchar(ch);
        fclose(fp);

        return 0;
    }

    程序打印以命令行参数（第二个）作为文件名的文件中，所有的数字。
